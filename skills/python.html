Example: Web Scraping with Beautiful Soup

import requests
from bs4 import BeautifulSoup
import pandas as pd

def scrape_tech_news():
    """Scrape technology headlines from a news website"""
    # Send HTTP request to the website
    url = "https://news.ycombinator.com/"
    response = requests.get(url)
    
    # Parse HTML content
    soup = BeautifulSoup(response.text, 'html.parser')
    
    # Extract news titles and links
    titles = soup.select('.titleline > a')
    scores = soup.select('.score')
    
    # Create data structure
    news_items = []
    
    for i in range(min(len(titles), len(scores))):
        item = {
            'title': titles[i].text,
            'link': titles[i]['href'],
            'points': int(scores[i].text.split()[0])
        }
        news_items.append(item)
    
    # Convert to DataFrame and sort by points
    df = pd.DataFrame(news_items)
    return df.sort_values('points', ascending=False)

# Example usage
top_tech_news = scrape_tech_news()
print(f"Found {len(top_tech_news)} news items")
print(top_tech_news.head(3))
